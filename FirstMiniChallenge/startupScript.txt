#!/bin/bash

sleep 20

WEBHOOK_URL="..."


HOSTNAME=$(hostname)
USER_NAME=$(whoami)
DATE_NOW=$(date '+%Y-%m-%d %H:%M:%S')
LOCAL_IP=$(hostname -I | awk '{print $1}')

MESSAGE=" Jetson wifi

Host: $HOSTNAME
Usuario: $USER_NAME
Fecha: $DATE_NOW
IP: $LOCAL_IP"

curl -H "Content-Type: application/json" \
     -X POST \
     -d "{\"content\":\"$MESSAGE\"}" \
     "$WEBHOOK_URL"
